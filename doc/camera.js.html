// This is a quick fix for this issue:
// https://stackoverflow.com/questions/52532614/total-canvas-memory-use-exceeds-the-maximum-limit-safari-12
// Note: There is a risk the canvas gets overwritten before it gets used. A
// better method would be to get image data from that canvas (or create one each
// time the function is called, but making sure the canvas are discarded once
// they are not useful anymore)
const snapCanvas = document.createElement('canvas');

/**
 * @param {HTMLVideoElement} videoElement
 * @return {HTMLCanvasElement} The snapped frame, in the form of a canvas
 */
export function snapFrame(videoElement) {
  snapCanvas.height = videoElement.videoHeight;
  snapCanvas.width = videoElement.videoWidth;
  const ctx = snapCanvas.getContext('2d');
  ctx.drawImage(videoElement, 0, 0, snapCanvas.width, snapCanvas.height);

  return snapCanvas;
}
