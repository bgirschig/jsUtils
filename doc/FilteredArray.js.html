/**
 * 'Smoothes' an array's values across frames, by applying a low pass filter on them.
 * The filter is 'inert': it only updates when getValues() is called.
 * Each frame, the array values get closer to their target values
 */
export default class FilteredArray {
  /**
   * @param {Number[]} start - The first state of the array. This must have the
   * desired size for the array
   * @param {Number} factor - How much to smooth the values
   * @param {boolean} skipZeros - If true, if a 'target' or 'previous' value is
   * 0, jump directly to the next value
   */
  constructor(start, factor = 5, skipZeros = true) {
    this._prev_value = start;
    this._target = start;
    this.factor = factor;
    this.skipZeros = skipZeros;
    /** Number of steps since the last target was set. This provides a
     * lightwheight way of checking if the 'transition' is done. (Compared to
     * computing the real distance to target) */
    this.stepCount = 0;
  }

  /**
   * Set the target values for this array
   * @param {Number} value
   */
  set target(value) {
    this.stepCount = 0;
    console.assert(
        value.length === this._prev_value.length,
        'FilteredArray target must be an array with same length');
    this._target = value;
  }

  /**
   * Returns the current filtered value.
   * @param {boolean} peek If true, returns the current value without updating
   * the filter
   * @return {Array}
   */
  getValue(peek = false) {
    if (peek) return this._prev_value;

    // Using map() helps keep the origina array kind: (regular, typed array,
    // etc...) without any extra logic.
    this._prev_value = this._prev_value.map((prev, idx) => {
      if (this.skipZeros && (this._target[idx] === 0 || prev === 0)) {
        // Don't interpolate to/from values that are exactly 0 if the
        // 'skipZeros' option is set.
        return this._target[idx];
      }
      return prev + (this._target[idx] - prev) / this.factor;
    });

    this.stepCount += 1;
    return this._prev_value;
  }
}
